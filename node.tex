\subsection*{Matrices of Nodes}

\textit{Create a set of nodes with a \textbf{matrix}:}\\
\entry{40mm}{\textbackslash matrix [matrix of nodes]\{}{create}\\
\entry{40mm}{\phantom{xx}1 \& 2 \& 3 \textbackslash\textbackslash}{basic syntax}\\
\entry{40mm}{\phantom{xx}4 \& 5 \& |red| 6 \}}{pass node opt\textquotesingle n}\\

\textit{Other matrix options $\in$:}
{\scriptsize 
\begin{multicols}{3}\begin{itemize}[leftmargin=0mm,label={}]
    \item draw
    \item anchor
    \item column sep
    \item row sep
    \item every odd col.
    \item every even col.
    \item row \# column \#
    \item above delimiter
    \item left delimiter
\end{itemize}\end{multicols}}
