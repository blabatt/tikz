\begin{tabularx}{5cm}{l l}
    \underline{style} & \underline{example val} \\

    anchor & west \\

    pos & .5 \\

    decoration & snake \\
    fill & green \\
    color & green\\

\code{[mystyle/.style=\{draw=\#1,fill=\#1!50\}]}\\
\code{[mystyle/.append style=blue!50]}\\

\subsection*{Data \& Control}
\textit{Wrap the following in } \texttt{\textbackslash tikzmath\{...\} }:\\
\entry{44mm}{\textbackslash a = 4*5+6;}{assignment}\\
\entry{44mm}{if \textbackslash k>170 then \{let \textbackslash c = blue;\}}{conditional}\\
\entry{44mm}{for \textbackslash x in \{\dots \}}{looping}\\
\entry{54mm}{function product(\textbackslash x,\textbackslash y)\{return\textbackslash x*\textbackslash y;\};}{func}\\


\textit{Low-level parsing is similar to \LaTeX :}\\
\entry{53mm}{\textbackslash def\textbackslash myarray\{\{1,"two",2+1,sin(\textbackslash i*5)\}\}}{array}\\
\entry{40mm}{\textbackslash pgfparse \{\textbackslash myarray[3]\}}{extract from}\\
\entry{40mm}{\textbackslash foreach \textbackslash i in\{0,...,6\}}{loop over}\\
\entry{40mm}{\textbackslash pgfmathparse\{sqrt(10)\}}{apply func}\\
\entry{40mm}{\textbackslash pgfmathparse\{pi\}}{constant $\pi$}\\
\entry{40mm}{\textbackslash pgfmathresult}{return result}\\
