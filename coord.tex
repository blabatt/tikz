\subsection*{Coordinate Systems}

\textit{Above examples make \say{implicit} reference to desired \say{coordinate system.} To do so explicitly: }
\begin{lstlisting}
(<csname> cs: <coordspec>)
\end{lstlisting}
\textit{where }\texttt{<csname>}\textit{ $\in$: }\texttt{canvas}, \texttt{xyz}, \texttt{canvas polar}, \texttt{barycentric}, \texttt{node}\textit{, or } \texttt{tangent}\textit{.}\\

\textit{Layer a new (scoped) canvas coordinate system:}\\
\code{\textbackslash begin\{scope\}[xshift=6cm]} ... \\

\textit{Use multiple cs\textquotesingle s with arbitrary complexity:}\\
\entry{43mm}{\textbackslash draw[](A) -{}- ( \$(A)!.85!(B)\$ );}{tween+node} \\
\entry{55mm}{\textbackslash node at(\{\$(A)!.5!(B)\$\} -| \{\$(C)!.5!(D)\$\})}{3cs\textquotesingle s}\\
